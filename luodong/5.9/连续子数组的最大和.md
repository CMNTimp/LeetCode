class Solution {
    public int maxSubArray(int[] nums) {
        int curMax = nums[0];
        int max = nums[0];
        for(int i=1;i<nums.length;i++){
            //要么是加入，要么是从i开始
            curMax = Math.max(curMax+nums[i],nums[i]);
            //记录最大值
            max = Math.max(curMax,max);
        }
        return max;
    }
}
